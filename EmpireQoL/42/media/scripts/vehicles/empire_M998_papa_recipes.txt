    /* EmpireQoL :: craftable parts for Papa_Chad's U.S. M998 Humvee.
       Papa's mod is loot-only (ArmyHangarMechanics / CrateMechanics /
       MechanicSpecial tables); KI5's M998 recipes make parts for a different
       truck. These recipes give the driven Humvee a native crafting path,
       priced on KI5's bullbar recipe (metal + welding). Metal parts only --
       glass (windshield/side windows) and door/seat variants stay loot. */

    craftRecipe Empire_M998_MakeBullBar
    {
        timedAction = Welding,
        time = 200,
        Tags = InHandCraft;Welding,
        category = Armoring,
        xpAward = MetalWelding:25,
        SkillRequired = MetalWelding:5,
        inputs
        {
            item 4 [Base.SmallSheetMetal],
            item 6 [Base.MetalPipe],
            item 2 [Base.Screws],
            item 9 [Base.BlowTorch],
            item 1 tags[WeldingMask] mode:keep,
        }
        outputs
        {
            item 1 Base.M998_Bull_Bar2,
        }
    }
